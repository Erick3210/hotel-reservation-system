CREATE TABLE Reserva (
    id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, -- Identificador único
    fecha_inicio VARCHAR2(255) NOT NULL, -- Fecha y hora de inicio de la reserva como String
    fecha_fin VARCHAR2(255) NOT NULL, -- Fecha y hora de fin de la reserva como String
    id_habitacion NUMBER NOT NULL, -- Identificador único de la habitación o sala reservada
    nombre_cliente VARCHAR2(255) NOT NULL, -- Nombre del cliente
    estado_reserva VARCHAR2(50) CHECK (estado_reserva IN ('confirmada', 'pendiente', 'cancelada')) NOT NULL, -- Estado de la reserva
    CONSTRAINT chk_fecha CHECK (fecha_inicio < fecha_fin) -- Validación de coherencia de fechas
);

SELECT * FROM Reserva;
